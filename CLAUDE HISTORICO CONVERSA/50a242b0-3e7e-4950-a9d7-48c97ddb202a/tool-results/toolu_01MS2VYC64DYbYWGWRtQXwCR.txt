  3522→    async function hmacHex(msg, secret, algo = 'SHA-256') {
  3523→      const enc = new TextEncoder();
  3524→      const key = await crypto.subtle.importKey('raw', enc.encode(secret), { name: 'HMAC', hash: algo }, false, ['sign']);
  3525→      const sig = await crypto.subtle.sign('HMAC', key, enc.encode(msg));
  3526→      return Array.from(new Uint8Array(sig)).map(b => b.toString(16).padStart(2, '0')).join('');
  3527→    }
  3528→    async function hmacB64(msg, secret) {
  3529→      const enc = new TextEncoder();
  3530→      const key = await crypto.subtle.importKey('raw', enc.encode(secret), { name: 'HMAC', hash: 'SHA-256' }, false, ['sign']);
  3531→      const sig = await crypto.subtle.sign('HMAC', key, enc.encode(msg));
  3532→      return btoa(String.fromCharCode(...new Uint8Array(sig)));
  3533→    }
  3534→    async function sha512hex(msg) {
  3535→      const enc = new TextEncoder();
  3536→      const buf = await crypto.subtle.digest('SHA-512', enc.encode(msg || ''));
  3537→      return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('');
  3538→    }

  3619→    async function initTsOffset() {
  3620→      if (TS_OFFSET_FETCHED) return;
  3621→      // Tenta buscar timestamp real do servidor via proxy
  3622→      const apis = [
  3623→        { url: 'https://api.mexc.com/api/v3/time', parse: d => d.serverTime },
  3624→        { url: 'https://api.binance.com/api/v3/time', parse: d => d.serverTime },
  3625→        { url: 'https://api.kucoin.com/api/v1/timestamp', parse: d => d.data },
  3626→        { url: 'https://api.gateio.ws/api/v4/spot/time', parse: d => d.server_time * 1000 }
  3627→      ];
  3628→      for (const api of apis) {
  3629→        try {
  3630→          const before = Date.now();
  3631→          const r = await apiFetch(api.url, { method: 'GET' }, true);
  3632→          if (!r) continue;
  3633→          if (!r.ok) continue;
  3634→          const d = await r.json();
  3635→          const serverTs = api.parse(d);
  3636→          if (serverTs && serverTs > 1000000000000 && serverTs < 2000000000000) {
  3637→            const latency = Math.round((Date.now() - before) / 2);
  3638→            GLOBAL_TS_OFFSET = (serverTs + latency) - Date.now();
  3639→            TS_OFFSET_FETCHED = true;
  3640→            console.log('[TS] Offset: ' + GLOBAL_TS_OFFSET + 'ms via ' + api.url + ' (ts=' + serverTs + ')');
  3641→            return;
  3642→          }
  3643→        } catch (e) { console.warn('[TS] ' + api.url + ' falhou:', errMsg(e)); continue }
  3644→      }
  3645→      // Fallback: se PC está em 2082 (timestamp > ano 2050 = 2524608000000ms)
  3646→      const wrongNow = Date.now();
  3647→      if (wrongNow > 2524608000000) {
  3648→        // Diferença em ms de 2082 para 2026 = ~56 anos
  3649→        GLOBAL_TS_OFFSET = -(wrongNow - 1740534000000);
  3650→        console.warn('[TS] Fallback offset: ' + GLOBAL_TS_OFFSET + 'ms');
  3651→      }
  3652→      TS_OFFSET_FETCHED = true;
  3653→    }
